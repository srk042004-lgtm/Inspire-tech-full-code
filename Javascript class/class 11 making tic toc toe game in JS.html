üéØ JavaScript Project 1: Tic-Tac-Toe Game

This project brings together almost everything you have learned: HTML structure, CSS layout (Grid), JavaScript variables, arrays, loops, conditionals, DOM manipulation, and event handling.

Demo: Play the Game

Player X's Turn

Part 1: Setting up the JavaScript Logic

The game logic relies on an **array** to track the state of the 9 cells, and several **functions** to handle user interaction and check for a winner.

A. Variables and Initial Setup
// Global Variables for Game State const statusDisplay = document.querySelector('#status'); const cells = document.querySelectorAll('.cell'); const resetButton = document.querySelector('#reset-btn'); let gameActive = true; let currentPlayer = "X"; // Array to track the state of the board: ["", "X", "O", "", "", "", "", "", ""] let gameState = ["", "", "", "", "", "", "", "", ""]; // Winning conditions: all possible lines of 3 cells const winningConditions = [ [0, 1, 2], [3, 4, 5], [6, 7, 8], // Rows [0, 3, 6], [1, 4, 7], [2, 5, 8], // Columns [0, 4, 8], [2, 4, 6] // Diagonals ]; // Messages displayed to the user const winMessage = (player) => `Player ${player} has won!`; const drawMessage = () => `Game ended in a draw!`; const currentPlayerTurn = (player) => `Player ${player}'s Turn`; statusDisplay.innerHTML = currentPlayerTurn(currentPlayer);
B. Core Function: Handling a Cell Click

This function is called when a user clicks on an empty cell. It updates the state, updates the DOM, and then checks for a winner.

function handleCellPlayed(clickedCell, clickedCellIndex) { // 1. Update the gameState array with the current player's mark gameState[clickedCellIndex] = currentPlayer; // 2. Update the visual DOM element clickedCell.innerHTML = currentPlayer; clickedCell.classList.add(currentPlayer); // Add class for styling (X or O color) } function handlePlayerChange() { // Ternary Operator (Condition ? ValueIfTrue : ValueIfFalse) currentPlayer = currentPlayer === "X" ? "O" : "X"; statusDisplay.innerHTML = currentPlayerTurn(currentPlayer); }
C. The Winner Checker Function (Using Loops)

This function loops through the `winningConditions` array and uses conditionals to see if any line has three matching marks.

function handleResultValidation() { let roundWon = false; // Loop through all 8 winning conditions for (let i = 0; i <= 7; i++) { const winCondition = winningConditions[i]; // Use destructuring to pull the three indices let a = gameState[winCondition[0]]; let b = gameState[winCondition[1]]; let c = gameState[winCondition[2]]; // If any cell in the line is empty, skip this check if (a === '' || b === '' || c === '') { continue; } // Check if all three cells match if (a === b && b === c) { roundWon = true; break; // Exit the loop immediately, we found a winner! } } if (roundWon) { statusDisplay.innerHTML = winMessage(currentPlayer); gameActive = false; return; } // Check for a Draw: if no winner and no empty cells left let roundDraw = !gameState.includes(""); if (roundDraw) { statusDisplay.innerHTML = drawMessage(); gameActive = false; return; } // If no win and no draw, switch players handlePlayerChange(); }

Part 2: Combining DOM and Events

A. The Main Click Handler

This function acts as the main handler for all clicks, ensuring the game rules are followed before calling the logic functions.

function handleCellClick(event) { const clickedCell = event.target; const clickedCellIndex = parseInt(clickedCell.getAttribute('data-index')); // Rule Check 1: Is the cell already played OR is the game over? if (gameState[clickedCellIndex] !== "" || !gameActive) { return; // Ignore the click } // Core execution handleCellPlayed(clickedCell, clickedCellIndex); handleResultValidation(); }
B. The Reset Function

Clears the board, resets the state variables, and allows the game to restart.

function handleRestartGame() { gameActive = true; currentPlayer = "X"; gameState = ["", "", "", "", "", "", "", "", ""]; statusDisplay.innerHTML = currentPlayerTurn(currentPlayer); // Clear all cell contents and classes using a loop (Array method: forEach) cells.forEach(cell => { cell.innerHTML = ""; cell.classList.remove('X'); cell.classList.remove('O'); }); }
C. Attaching Listeners

We attach the `handleCellClick` listener to **every single cell** using the `forEach` array method, and attach the `handleRestartGame` to the button.

// Attach the click handler to all 9 cells cells.forEach(cell => cell.addEventListener('click', handleCellClick)); // Attach the restart handler to the button resetButton.addEventListener('click', handleRestartGame);

Putting It All Together (Full Script)

To make the demo above functional, this is the full JavaScript block that should be placed inside `<script>` tags at the end of the HTML body.

// FULL JAVASCRIPT CODE TO BE PLACED IN A <script> BLOCK const statusDisplay = document.querySelector('#status'); const cells = document.querySelectorAll('.cell'); const resetButton = document.querySelector('#reset-btn'); let gameActive = true; let currentPlayer = "X"; let gameState = ["", "", "", "", "", "", "", "", ""]; const winningConditions = [ [0, 1, 2], [3, 4, 5], [6, 7, 8], [0, 3, 6], [1, 4, 7], [2, 5, 8], [0, 4, 8], [2, 4, 6] ]; const winMessage = (player) => `Player ${player} has won!`; const drawMessage = () => `Game ended in a draw!`; const currentPlayerTurn = (player) => `Player ${player}'s Turn`; statusDisplay.innerHTML = currentPlayerTurn(currentPlayer); function handleCellPlayed(clickedCell, clickedCellIndex) { gameState[clickedCellIndex] = currentPlayer; clickedCell.innerHTML = currentPlayer; clickedCell.classList.add(currentPlayer); } function handlePlayerChange() { currentPlayer = currentPlayer === "X" ? "O" : "X"; statusDisplay.innerHTML = currentPlayerTurn(currentPlayer); } function handleResultValidation() { let roundWon = false; for (let i = 0; i <= 7; i++) { const winCondition = winningConditions[i]; let a = gameState[winCondition[0]]; let b = gameState[winCondition[1]]; let c = gameState[winCondition[2]]; if (a === '' || b === '' || c === '') { continue; } if (a === b && b === c) { roundWon = true; break; } } if (roundWon) { statusDisplay.innerHTML = winMessage(currentPlayer); gameActive = false; return; } let roundDraw = !gameState.includes(""); if (roundDraw) { statusDisplay.innerHTML = drawMessage(); gameActive = false; return; } handlePlayerChange(); } function handleCellClick(event) { const clickedCell = event.target; const clickedCellIndex = parseInt(clickedCell.getAttribute('data-index')); if (gameState[clickedCellIndex] !== "" || !gameActive) { return; } handleCellPlayed(clickedCell, clickedCellIndex); handleResultValidation(); } function handleRestartGame() { gameActive = true; currentPlayer = "X"; gameState = ["", "", "", "", "", "", "", "", ""]; statusDisplay.innerHTML = currentPlayerTurn(currentPlayer); cells.forEach(cell => { cell.innerHTML = ""; cell.classList.remove('X'); cell.classList.remove('O'); }); } cells.forEach(cell => cell.addEventListener('click', handleCellClick)); resetButton.addEventListener('click', handleRestartGame);
